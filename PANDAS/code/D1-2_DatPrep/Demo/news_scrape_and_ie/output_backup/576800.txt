การบินไทย  อนุมัติแผนจัดซื้อเครื่องบินใหม่ 28 ลำ เล็งชง ครม. เห็นชอบ พ.ย.นี้ พร้อมเปิดเส้นทางบินใหม่ กรุงเทพฯ-เวียนนา
ร.อ.มนตรี จำเนียร รองกรรมการผู้อำนวยการใหญ่ บริษัท การบินไทย จำกัด (มหาชน) เปิดเผยว่าในการประชุมคณะกรรมการบริษัทฯล่าสุดเมื่อวันที่ 12 ก.ค.ที่ผ่านมาว่า ที่ประชุมอนุมัติแผนยุทธ์ศาสตร์ 5 ปี เรื่องการให้จัดหาเครื่องบินใหม่แทนเครื่องบินเก่าจำนวน 28 ลำ
พร้อมทั้งสั่งให้ศึกษารายละเอียด ทั้งข้อเสนอผู้ผลิตและผู้ใช้เช่าเครื่องบิน แผนการเงิน รายได้ที่จะเกิดขึ้นในอนาคตโดย คาดว่าในเดือน ส.ค. นี้จะเสนอแบบเครื่องบินและราคาเครื่องบินทั้งหมดให้คณะกรรมการพิจารณาอนุมัติ ทั้งนี้หากได้รับความเห็นชอบจะเสนอเรื่องไปยังกระทรวงคมนาคมและหน่วยงานที่เกี่ยวข้องและคาดว่าจะสามารถเสนอคณะรัฐมนตรี หรือครม.อนุมัติหรือในช่วงเดือนพ.ย.2560
ขณะเดียวกัน ที่ประชุมได้อนุมัติให้เปิดเส้นทางบินใหม่คือ กรุงเทพฯ-เวียนนา 4 เที่ยวบินต่อสัปดาห์ ซึ่งในอนาคตเชื่อว่าจะเป็นทางที่มีศักยภาพและมีแหล่งท่องเที่ยวจำนวนมากรวมทั้งจะเป็นจุดเครือข่ายด้านการบินของการบินไทยในยุโรปตะวันออกอีกด้วย
